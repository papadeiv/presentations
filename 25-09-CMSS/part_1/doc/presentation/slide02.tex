        \footnotesize
        \begin{columns}[T]
                \column{0.5\textwidth}
                
                \begin{equation*}
                        dx_{t} = -\newprime{V}(x)dt + \sqrt{2D}dW_{t}\,,
                \end{equation*}

                \uncover<2->{
                        \begin{equation*}
                            \partial_{t}\rho(x,t)=\mathcal{L}\rho(x,t)\,.
                        \end{equation*}
                }

                \uncover<3->{
                        In the \textbf{time asymptotic limit}:
                        \begin{equation*}
                            \rho(x,t)\to\rho_{*}(x)\,,\quad0\approx \mathcal{L}\rho_{*} = \newprime{(V\rho_{*})} + D \pprime{\rho_{*}}\,,
                        \end{equation*}
                }

                \vspace{-0.5cm}
                \uncover<4->{
                        \begin{equation*}
                            \rho_{*}(x) = N\,\text{exp}\bigg(-\frac{V(x)}{D}\bigg)\,.
                        \end{equation*}
                }

                \medskip
                \uncover<6->{
                        \textit{Mean first passage time (MFPT)} in the \textbf{weak noise limit} $(D<<1)$:
                        \begin{equation*}
                                \tau_{\text{esc}}\approx \frac{2\pi}{\sqrt{\pprime{V}(a)|\pprime{V}(b)|}}\,\text{exp}\bigg(\frac{V(b)-V(a)}{D}\bigg)\,.
                        \end{equation*}
                }
